  • setter/getter

    var obj = {
  name: 'bigBoy',
  get sex() {
    if(this.sexx) {
      return 'man';
    } else {
      return 'man';
    }
  },
  set sex(val) {
    this.sexx = val;
  }
};
obj.sex = 'woman';     // obj既有set又有get，所以该操作被允许
console.log(obj.sex);  // woman
